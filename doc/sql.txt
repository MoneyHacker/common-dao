CREATE TABLE `sys_dict` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '字典内码',
  `parent_id` bigint(20) DEFAULT '0' COMMENT '父级id',
  `parent_ids` varchar(200) COLLATE utf8_bin DEFAULT NULL COMMENT '所有上级',
  `sort` smallint(6) DEFAULT '0' COMMENT '排序',
  `dict_code` varchar(120) COLLATE utf8_bin DEFAULT NULL COMMENT '字典编号',
  `dict_name` varchar(120) COLLATE utf8_bin DEFAULT NULL COMMENT '字典名称',
  `dict_value` varchar(120) COLLATE utf8_bin DEFAULT NULL COMMENT '字典值',
  `editable` tinyint(4) DEFAULT NULL COMMENT '是否可编辑',
  `active` tinyint(4) NOT NULL DEFAULT '0' COMMENT '是否有效 0:有效 -1:无效',
  `create_user_id` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '创建人ID',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_user_id` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '修改人ID',
  `update_time` datetime DEFAULT NULL COMMENT '修改时间',
  `remark` varchar(200) COLLATE utf8_bin DEFAULT NULL COMMENT '备注',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8 COLLATE=utf8_bin ROW_FORMAT=COMPACT;

CREATE TABLE `sys_user` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '登陆内码',
  `company_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '归属公司',
  `office_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '所属部门',
  `password` varchar(100) COLLATE utf8_bin DEFAULT '94072185974193097cb426d1284afbec4f8207a5c2c5c8e9ba056740ddffc355',
  `salt` varchar(120) COLLATE utf8_bin DEFAULT '58_adv_cms_%$#@ds520' COMMENT '随机数',
  `user_name` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '昵称',
  `real_name` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '真实姓名',
  `user_code` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '用户编号',
  `sex` tinyint(4) DEFAULT NULL COMMENT '性别 0:未知 1:男 2:女',
  `user_born` datetime DEFAULT NULL COMMENT '生日',
  `from_source` varchar(20) COLLATE utf8_bin DEFAULT NULL COMMENT '来源 WECHAT  PC  MOBILE',
  `img_url` varchar(120) COLLATE utf8_bin DEFAULT NULL COMMENT '图片地址',
  `cell_phone` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '电话',
  `email` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '邮箱',
  `active` tinyint(4) NOT NULL DEFAULT '0' COMMENT '是否有效 0:有效 -1:无效',
  `create_user_id` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '创建人ID',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_user_id` varchar(40) COLLATE utf8_bin DEFAULT NULL COMMENT '修改人ID',
  `update_time` datetime DEFAULT NULL COMMENT '修改时间',
  `remark` varchar(200) COLLATE utf8_bin DEFAULT NULL COMMENT '备注',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8 COLLATE=utf8_bin ROW_FORMAT=COMPACT;
